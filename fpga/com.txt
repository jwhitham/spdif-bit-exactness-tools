-- com_data bits

 15 downto 14 set the command
   00 = do nothing
   01 = set mode
     8          -> pre-emphasis value (0 or 1)
     7          -> pre-emphasis set
     6 downto 3 -> mode setting value (0000 - 1010, as in mode_definitions.vhdl)
     2          -> mode set (enable remote control)
     1          -> mode clear (return to manual control)
     0          -> reset error (same as button C6)
   10 = set adjust 1
     9 downto 0 -> value
   11 = set adjust 2
     9 downto 0 -> value
